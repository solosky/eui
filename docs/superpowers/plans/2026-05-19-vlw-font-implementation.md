**Goal:** Implement Processing `.vlw` font format reading and rendering in `eui_font_vlw.c`

**Architecture:** Add three helper functions (big-endian int32 reader, binary-search glyph lookup, bitmap offset calculator) and implement the three existing stub functions (get_char_width, get_str_width, draw_char). VLW 8bpp grayscale bitmaps are thresholded at 128 for ink/background decision.

**Tech Stack:** C99, CMake, no external dependencies. Standard `<stdint.h>` for `int32_t`.

```python
#!/usr/bin/env python3
"""Generate a test VLW font for unit testing.
Format: Processing .vlw binary, all ints big-endian.
8 glyphs (A-H), 8x8 each, at 12px size.
"""
import struct

def be32(v):
    return struct.pack('>i', v)

def bdf_row_to_8bpp(byte, width=8):
    """Convert 1bpp BDF row byte to 8 8bpp bytes. 1=ink(0), 0=bg(255)"""
    return bytes(0 if (byte >> (7-i)) & 1 else 255 for i in range(width))

# BDF glyph bitmaps (8x8) for chars A-H, matching eui_font_builtin.c
bdf_glyphs = {
    'A': [0x18, 0x3C, 0x66, 0x7E, 0x66, 0x66, 0x66, 0x00],
    'B': [0x7C, 0x66, 0x66, 0x7C, 0x66, 0x66, 0x7C, 0x00],
    'C': [0x3C, 0x66, 0x60, 0x60, 0x60, 0x66, 0x3C, 0x00],
    'D': [0x78, 0x6C, 0x66, 0x66, 0x66, 0x6C, 0x78, 0x00],
    'E': [0x7E, 0x60, 0x60, 0x7C, 0x60, 0x60, 0x7E, 0x00],
    'F': [0x7E, 0x60, 0x60, 0x7C, 0x60, 0x60, 0x60, 0x00],
    'G': [0x3C, 0x66, 0x60, 0x6E, 0x66, 0x66, 0x3C, 0x00],
    'H': [0x66, 0x66, 0x66, 0x7E, 0x66, 0x66, 0x66, 0x00],
}

glyph_count = 8
version = 11
font_size = 12
ascent = 10
descent = 2
glyph_w = 8
glyph_h = 8
set_width = 8

data = bytearray()

# Header (24 bytes)
data += be32(glyph_count)   # [0-3]
data += be32(version)       # [4-7]
data += be32(font_size)     # [8-11]
data += be32(0)             # [12-15] mboxY (ignored)
data += be32(ascent)        # [16-19]
data += be32(descent)       # [20-23]

# Glyph headers (28 bytes each, sorted A-H)
for ch in sorted(bdf_glyphs.keys()):
    data += be32(ord(ch))   # [0-3]  value (Unicode code point)
    data += be32(glyph_h)   # [4-7]  height
    data += be32(glyph_w)   # [8-11] width
    data += be32(set_width) # [12-15] setWidth
    data += be32(8)         # [16-19] topExtent
    data += be32(0)         # [20-23] leftExtent
    data += be32(0)         # [24-27] padding

# Glyph bitmaps (widht * height bytes each, sequential)
for ch in sorted(bdf_glyphs.keys()):
    rows = bdf_glyphs[ch]
    for row in rows:
        data += bdf_row_to_8bpp(row, 8)

# Convert to C array
hex_bytes = ', '.join(f'0x{b:02X}' for b in data)

print(f'''/* Auto-generated test VLW font (Processing .vlw format)
 * 8 glyphs A-H, 8x8 pixels, 8bpp grayscale.
 * Generated by tools/gen_test_vlw.py
 */
#ifndef TEST_VLW_FONT_H
#define TEST_VLW_FONT_H

#include <stdint.h>

#define TEST_VLW_GLYPH_COUNT {glyph_count}
#define TEST_VLW_ASCENT {ascent}
#define TEST_VLW_DESCENT {descent}
#define TEST_VLW_SIZE {font_size}

static const uint8_t test_vlw_font_data[] = {{
    {hex_bytes}
}};

#endif /* TEST_VLW_FONT_H */
''')
```

```c
#include "eui/eui_font.h"
#include "eui_font_internal.h"

#define VLW_HEADER_SIZE 24
#define VLW_GLYPH_HEADER_SIZE 28

static int32_t vlw_read_int32(const uint8_t *p)
{
    return ((int32_t)p[0] << 24) | ((int32_t)p[1] << 16) |
           ((int32_t)p[2] << 8)  |  (int32_t)p[3];
}

/*
 * Binary search glyph table for Unicode code point c.
 * Returns pointer to glyph header (28 bytes) or NULL if not found.
 * Glyph table is sorted by code point in ascending order.
 */
static const uint8_t* find_glyph(const eui_font_t *font, char c)
{
    const uint8_t *p = font->data;
    if (!p) return NULL;

    int32_t count = vlw_read_int32(p);
    if (count <= 0) return NULL;

    const uint8_t *glyphs = p + VLW_HEADER_SIZE;
    int lo = 0, hi = count - 1;

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        const uint8_t *gh = glyphs + mid * VLW_GLYPH_HEADER_SIZE;
        int32_t v = vlw_read_int32(gh);
        if ((int32_t)(uint8_t)c == v) return gh;
        if ((int32_t)(uint8_t)c < v) hi = mid - 1;
        else lo = mid + 1;
    }
    return NULL;
}

/*
 * Find start of bitmap data for glyph at target_idx.
 * Glyph bitmaps are concatenated sequentially after glyph headers.
 */
static const uint8_t* glyph_bitmap(const eui_font_t *font, int target_idx)
{
    const uint8_t *p = font->data;
    int32_t count = vlw_read_int32(p);
    const uint8_t *glyphs = p + VLW_HEADER_SIZE;
    const uint8_t *bitmap_base = glyphs + count * VLW_GLYPH_HEADER_SIZE;
    int32_t offset = 0;
    for (int i = 0; i < target_idx; i++) {
        const uint8_t *gh = glyphs + i * VLW_GLYPH_HEADER_SIZE;
        int32_t w = vlw_read_int32(gh + 4);
        int32_t h = vlw_read_int32(gh + 8);
        offset += w * h;
    }
    return bitmap_base + offset;
}

/*
 * Find glyph index for Unicode code point c.
 * Returns index (0..count-1) or -1 if not found.
 */
static int find_glyph_index(const eui_font_t *font, char c)
{
    const uint8_t *p = font->data;
    if (!p) return -1;

    int32_t count = vlw_read_int32(p);
    if (count <= 0) return -1;

    const uint8_t *glyphs = p + VLW_HEADER_SIZE;
    int lo = 0, hi = count - 1;

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        const uint8_t *gh = glyphs + mid * VLW_GLYPH_HEADER_SIZE;
        int32_t v = vlw_read_int32(gh);
        if ((int32_t)(uint8_t)c == v) return mid;
        if ((int32_t)(uint8_t)c < v) hi = mid - 1;
        else lo = mid + 1;
    }
    return -1;
}

uint8_t eui_font_vlw_get_char_width(const eui_font_t *font, char c)
{
    const uint8_t *g = find_glyph(font, c);
    if (!g) return 0;
    return (uint8_t)vlw_read_int32(g + 12);
}

uint16_t eui_font_vlw_get_str_width(const eui_font_t *font, const char *str)
{
    uint16_t w = 0;
    while (*str) {
        w += eui_font_vlw_get_char_width(font, *str);
        str++;
    }
    return w;
}

uint8_t eui_font_vlw_draw_char(const eui_font_t *font, char c,
                                uint8_t *buf, uint16_t buf_stride,
                                uint8_t color_depth)
{
    int idx = find_glyph_index(font, c);
    if (idx < 0) return 0;

    const uint8_t *p = font->data;
    const uint8_t *glyphs = p + VLW_HEADER_SIZE;
    const uint8_t *gh = glyphs + idx * VLW_GLYPH_HEADER_SIZE;

    int32_t w = vlw_read_int32(gh + 4);
    int32_t h = vlw_read_int32(gh + 8);
    int32_t x_advance = vlw_read_int32(gh + 12);

    if (w <= 0 || h <= 0) return (uint8_t)x_advance;

    const uint8_t *bitmap = glyph_bitmap(font, idx);

    for (int32_t row = 0; row < h; row++) {
        for (int32_t col = 0; col < w; col++) {
            uint8_t pixel = bitmap[row * w + col];
            /* pixel 0=ink(opaque), 255=bg(transparent), threshold at 128 */
            if (pixel < 128) {
                if (color_depth == 1) {
                    buf[row * buf_stride + col / 8] |= (1u << (7 - (col % 8)));
                } else {
                    buf[row * buf_stride + col] = 0xFF;
                }
            }
        }
    }

    return (uint8_t)x_advance;
}
```
